/* cores do quadrado */
GLfloat r=1.0, g=0.5, b=0.0;
GLfloat a, u, c;

int main(int argc, char** argv){
    glutInit(&argc, argv);
    glutInitDisplayMode (GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize (256, 256); 
    glutInitWindowPosition (100, 100); 
    glutCreateWindow (argv[0]);
    // inicia um temporizador. após 33ms ativa a funcao timer
    glutTimerFunc(33, timer, 1);
    glClearColor(1.0, 1.0, 1.0, 0.0);
    glShadeModel (GL_FLAT);
    glOrtho (0, 1, 0, 1, -1 ,1);
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
void timer(int value){
    r=r+0.01;
    g=g+0.01;
    b=b+0.01;

    a=r+0.5; 
    u=g+0.5; 
    c=b+0.5;

    if(a>1) a-=1;
    if(u>1) u-=1;
    if(c>1) c-=1;

    if(r>1) r=0;
    if(g>1) g=0;
    if(b>1) b=0;
    glutPostRedisplay();
    glutTimerFunc(33, timer, 1);
}

void display(void){
    int i;
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f (r, g, b);
    glBegin(GL_POLYGON);
    glVertex2f(0.1,0.1);
    glVertex2f(0.5,0.1);
    glVertex2f(0.5,0.5);
    glVertex2f(0.1,0.5);
    glEnd();

    glColor3f (a, u, c);
    glBegin(GL_POLYGON);
    glVertex2f(0.25,0.25);

#include <GL/glut.h>
#include <stdlib.h>
#include <stdio.h>
#include <math.h>

void display(void);
void timer(int);
void keyboard(unsigned char key, int x, int y);

GLfloat rad(int teta){return (3.14159265359*teta)/180.;}
/* cores do quadrado */
GLfloat r[4],   g[4],   b[4];
int h[4] = {0.0,50.0,100.0,150.0};

int main(int argc, char** argv){
    glutInit(&argc, argv);
    glutInitDisplayMode (GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize (256, 256); 
    glutInitWindowPosition (100, 100); 
    glutCreateWindow (argv[0]);
    // inicia um temporizador. após 33ms ativa a funcao timer
    glutTimerFunc(33, timer, 1);
    glClearColor(1.0, 1.0, 1.0, 0.0);
    //glShadeModel (GL_FLAT);
    glOrtho (0, 1, 0, 1, -1 ,1);
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
void timer(int value){


    for(int j=0; j<4; j++){     

        h[j] = h[j] + 10.;
        if(h[j]>=360) h[j] = h[j]%360; 

        if(h[j]>=0 && h[j]<120){            
            b[j] = .25;
            r[j] = .5 + .25*cos(rad(h[j]))/cos(rad(60)-rad(h[j]));
            g[j] = 1.5 - (r[j] + .25);
        }else if(h[j]<240){
            r[j] = .25;
            g[j] = .25 + .25*cos(rad(h[j])-rad(120))/cos(rad(180)-rad(h[j]));
            b[j] = 1.5 - (.25 + g[j]);
        }else if(h[j]<360){
            g[j] = .25;
            b[j] = .5 + .25*cos(rad(h[j])-rad(240))/cos(rad(300)-rad(h[j]));
            r[j] = 1.5 - (.25 + b[j]);
        }
    while(r[j] > 1) r[j] = r[j] - 1.;
    while(g[j] > 1) g[j] = g[j] - 1.;
    while(b[j] > 1) b[j] = b[j] - 1.;

    }
    glutPostRedisplay();
    glutTimerFunc(33, timer, 1);
}

void display(void){
    int i;
    glClear(GL_COLOR_BUFFER_BIT);
    glBegin(GL_POLYGON);
    glColor3f (r[0], g[0], b[0]);
    glVertex2f(0.1,0.1);
    glColor3f (r[1], g[1], b[1]);
    glVertex2f(0.9,0.1);
    glColor3f (r[2], g[2], b[2]);
    glVertex2f(0.9,0.9);
    glColor3f (r[3], g[3], b[3]);
    glVertex2f(0.1,0.9);
    glEnd();

    glFlush();

#include <GL/glut.h>
#include <stdlib.h>
#include <math.h>

void init(void);
void display(void);
void setPixel(int x, int y);
void bresenham(int x1, int y1, int x2, int y2);
int Sinal(int x){ if(x >= 0) return 1; else return -1;}

int main(int argc, char** argv){
    glutInit(&argc, argv);
    glutInitDisplayMode (GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize (256, 256); 
    glutInitWindowPosition (100, 100); 
    glutCreateWindow ("Desenhando uma linha");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
void setPixel(int x, int y){  
    glColor3f (0.0, 1.0, 0.0);
    glBegin(GL_POINTS);
    glVertex2i(x,y);
    glEnd();
}
void bresenham(int x1, int y1, int x2, int y2){
    int Temp, new_e, Troca;     
    int x = x1;
    int y = y1;
    int Delta_x = abs(x2 - x1);
    int Delta_y = abs(y2 - y1);
    int s1 = Sinal(x2 - x1);
    int s2 = Sinal(y2 - y1);
    if(Delta_y > Delta_x){
        Temp = Delta_x;
        Delta_x = Delta_y;
        Delta_y = Temp;
        Troca = 1;
    }
    else{
       Troca = 0;
    }
    new_e = 2*Delta_y - Delta_x;
    for(int i=1; i<= Delta_x; i++){
        setPixel(x,y);
        while (new_e >= 0){
            if(Troca == 1){
                x = x + s1;
            }
            else{
                y = y + s2;
            }
            new_e = new_e - 2*Delta_x;
        }

        if(Troca == 1){
            y = y + s2;
        }
        else{
            x = x + s1;
        }
        new_e = new_e + 2*Delta_y;
    }
}

void init(void){
    glClearColor(1.0, 1.0, 1.0, 1.0);
    glOrtho (0, 256, 0, 256, -1 ,1);
}

void display(void){
    glClear(GL_COLOR_BUFFER_BIT);
    bresenham(40,200,200,10);
    glFlush();
}

int main(int argc, char** argv){
    glutInit(&argc, argv);
    glutInitDisplayMode (GLUT_SINGLE | GLUT_RGB);

void setPixel(int x, int y){  
    glColor3f (0.0, 0.0, 1.0);
    glBegin(GL_POINTS);
    glVertex2i(x,y);
    glEnd();
}
void pontosDaCircunferencia(int x, int y, int x0, int y0){
    setPixel(x+x0,y+y0);
    setPixel(-x+x0,y+y0);
    setPixel(x+x0,-y+y0);
    setPixel(-x+x0,-y+y0);
}
void bresenham(int x0, int y0, int raio){
    int x = 0;
    int y = raio;
    int d = 1 - raio;
    pontosDaCircunferencia(x,y,x0,y0);
    while(y > x){
        if(d < 0){
            d = d + 2*x + 3;
            x = x + 1;
        }
        else{
            d = d + 2*(x-y) + 5;
            x = x + 1;
            y = y - 1;
        }
        pontosDaCircunferencia(x,y,x0,y0);
    }
    y = 0;
    x = raio;
    d = 1 - raio;
    pontosDaCircunferencia(x,y,x0,y0);
    while(x > y){
        if(d < 0){
            d = d + 2*y + 3;
            y = y + 1;
        }
        else{
            d = d + 2*(y-x) + 5;
            y = y + 1;
            x = x - 1;
        }
        pontosDaCircunferencia(x,y,x0,y0);
    }
}

void init(void){
  glClearColor(1.0, 1.0, 1.0, 1.0);
  glOrtho (0, 256, 0, 256, -1 ,1);
}

void display(void){
    glClear(GL_COLOR_BUFFER_BIT);
    setPixel(128,128);
    bresenham(128,128,50);
    glFlush();
}

            g_b[n]=(GLfloat)rand()/(RAND_MAX+1.0);
            b_b[n]=(GLfloat)rand()/(RAND_MAX+1.0);
            glutPostRedisplay();
        }else if (state == GLUT_DOWN && printColor == 'f') {
            if( x>=30 & x<=113 & y>=143 & y<=226 ){
                n = 2;
            }else if ( x>=143 & x<=226 & y>=143 & y<=226 ){
                n = 3;
            }else if ( x>=30 & x<=113 & y>=30 & y<=113 ){
                n = 0;
            }else if ( x>=143 & x<=226 & y>=30 & y<=113 ){
                n = 1;
            }else break;
            r_f[n]=(GLfloat)rand()/(RAND_MAX+1.0);
            g_f[n]=(GLfloat)rand()/(RAND_MAX+1.0);
            b_f[n]=(GLfloat)rand()/(RAND_MAX+1.0);
            glutPostRedisplay();        
        }
    break;
  }
}
